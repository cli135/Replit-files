"class Solution {\n    public boolean isCovered(int[][] ranges, int left, int right) {\n        if (left <= right) {\n            // search for left in ranges\n            for (int[] range : ranges) {\n                if (range.length >= 2 && range[0] <= left && left <= range[1]) {\n                        // found it, keep on searching\n                        return isCovered(ranges, left + 1, right);\n                }\n            }\n            // nope not found this one\n            return false;\n        }\n        else {\n            // yep we made it to the end, finding everything!\n            return true;\n        }\n    }\n}"
